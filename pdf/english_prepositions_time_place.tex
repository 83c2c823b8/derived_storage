\section*{Prepositions for Time}

\begin{center}
\begin{tabular}{|c|l|l|}
\hline
\textbf{Preposition} & \textbf{Usage} & \textbf{Examples} \\
\hline
at & Precise time / point & at 7:00, at noon, at night, at Christmas \\
\hline
on & Specific day / date & on Monday, on December 25, on my birthday \\
\hline
in & Month / year / period & in December, in 2025, in the morning \\
\hline
by & Deadline & by Friday, by tomorrow \\
\hline
before & Earlier than & before the meeting \\
\hline
after & Later than & after lunch \\
\hline
during & Within a period & during the summer \\
\hline
for & Length of time & for three days \\
\hline
since & Starting point & since 2020 \\
\hline
until & End point & until midnight \\
\hline
\end{tabular}
\end{center}
